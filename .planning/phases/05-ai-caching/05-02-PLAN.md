```python
# --- Cache hit / miss via pipeline logic simulation ---

def test_two_condition_cache_hit_both_required():
    """If industry is set but hash doesn't match → NOT a cache hit (stale hash)."""
    desc = "A payments company"
    correct_hash = compute_description_hash(desc)
    stale_hash = "a" * 64
    # Simulating pipeline condition: hash match AND industry not None
    is_hit_correct = (correct_hash == correct_hash) and ("FinTech" is not None)
    is_hit_stale   = (stale_hash == correct_hash) and ("FinTech" is not None)
    assert is_hit_correct is True
    assert is_hit_stale is False

def test_partial_write_not_treated_as_cache_hit():
    """Hash matches but industry IS None → partial write → must re-analyze."""
    desc = "An AI tool for developers"
    computed_hash = compute_description_hash(desc)
    stored_hash = computed_hash  # hash was written, but analysis failed
    stored_industry = None       # analysis failed before writing industry
    is_cache_hit = (stored_hash == computed_hash) and (stored_industry is not None)
    assert is_cache_hit is False
```

```python
import pytest
from sqlalchemy import inspect, text
from sqlmodel import create_engine
from sqlmodel.pool import StaticPool


def test_migration_adds_column():

    with engine.connect() as conn:
        conn.execute(text(
            "CREATE TABLE company ("
            "id INTEGER PRIMARY KEY, company_name TEXT, description TEXT, "
            "website TEXT, industry TEXT, business_model TEXT, summary TEXT, use_case TEXT"
            ")"
        ))
        conn.commit()
    # Run migration
    from scripts.migrate_add_hash import _migrate_engine
    _migrate_engine(engine)
    insp = inspect(engine)
    cols = [c["name"] for c in insp.get_columns("company")]
    assert "description_hash" in cols


def test_migration_is_idempotent():
    """Running migrate() twice doesn't raise."""
    from sqlmodel import create_engine
    from sqlmodel.pool import StaticPool
    from sqlalchemy import text
    engine = create_engine("sqlite://", connect_args={"check_same_thread": False}, poolclass=StaticPool)
    with engine.connect() as conn:
        conn.execute(text(
            "CREATE TABLE company ("
            "id INTEGER PRIMARY KEY, company_name TEXT, description TEXT, "
            "website TEXT, industry TEXT, business_model TEXT, summary TEXT, "
            "use_case TEXT, description_hash TEXT"
            ")"
        ))
        conn.commit()
    from scripts.migrate_add_hash import _migrate_engine
    _migrate_engine(engine)  # column already exists — must not raise
```
