**File**: `tests/test_api.py`

**Goal**: Six test cases covering the full API surface: empty list, populated list, single-record retrieval, 404 on missing ID, 422 on non-integer path param, and presence of all four AI-generated fields in the response.

### Critical Setup Note

`app/config.py` executes `settings = Settings()` at **module import time**, which requires `OPENAI_API_KEY` to be present in the environment. `app/main.py` imports `app/database.py` which is imported by `app/config.py` indirectly through the module graph. To prevent a `pydantic.ValidationError` before any test runs, the env var must be injected with `os.environ.setdefault` **before** the first `from app.*` import on any line of this file. The two-line guard at the top of the file is non-negotiable.

### Full Implementation

```python
"""
FastAPI endpoint tests for Phase 4.

Test isolation strategy:
  - Each test gets a fresh in-memory SQLite database via the `session` fixture.
  - FastAPI's dependency injection is overridden so all routes use the test
    session instead of the production engine.
  - `app.dependency_overrides` is cleared after every test via `yield` + cleanup
    in the `client` fixture to prevent state leaking between tests.
  - `data/companies.db` is never read or written by this suite.

OPENAI_API_KEY guard (top of file, before any app import):
  app/config.py creates `settings = Settings()` at module level. Importing
  app.main triggers this chain: app.main -> app.database -> (app is a package
  that re-exports config) -> Settings() -> requires OPENAI_API_KEY.
  Setting a dummy value before importing prevents ValidationError at collection
  time. os.environ.setdefault is safe — it will not override a real key that is
  already set in the environment.
"""
import os

os.environ.setdefault("OPENAI_API_KEY", "test-key-not-used")

import pytest
from sqlmodel import Session, SQLModel, create_engine
from sqlmodel.pool import StaticPool
from starlette.testclient import TestClient

from app.database import get_db
from app.main import app
from app.models import Company


# ---------------------------------------------------------------------------
# Fixtures
# ---------------------------------------------------------------------------


@pytest.fixture(name="session")
def session_fixture():
    """
    Yield a SQLModel Session backed by a fresh in-memory SQLite database.

    StaticPool keeps the same in-memory database alive for the entire session
    lifetime — without it, each new connection would get an empty database
    because SQLite's in-memory mode creates a new DB per connection.
    """
    engine = create_engine(
        "sqlite://",
        connect_args={"check_same_thread": False},
        poolclass=StaticPool,
    )
    SQLModel.metadata.create_all(engine)
    with Session(engine) as session:
        yield session


@pytest.fixture(name="client")
def client_fixture(session: Session):
    """
    Yield a Starlette TestClient whose routes use the in-memory test session.

    The override is registered before the client is yielded and cleared
    unconditionally after the test completes (even on failure) via the
    fixture teardown.
    """
    def get_session_override():
        return session

    app.dependency_overrides[get_db] = get_session_override
    client = TestClient(app)
    yield client
    app.dependency_overrides.clear()


# ---------------------------------------------------------------------------
# Helper
# ---------------------------------------------------------------------------


def _make_company(
    name: str,
    description: str = "A great startup",
    website: str = "https://example.com",
    industry: str | None = None,
    business_model: str | None = None,
    summary: str | None = None,
    use_case: str | None = None,
) -> Company:
    """Construct an unsaved Company instance with sensible defaults."""
    return Company(
        company_name=name,
        description=description,
        website=website,
        industry=industry,
        business_model=business_model,
        summary=summary,
        use_case=use_case,
    )


# ---------------------------------------------------------------------------
# API-01: GET /companies
# ---------------------------------------------------------------------------


def test_get_companies_empty(client: TestClient):
    """
    GET /companies returns an empty JSON array when the database has no rows.
    Validates API-01: the endpoint exists, returns 200, and returns a list.
    """
    response = client.get("/companies/")
    assert response.status_code == 200
    assert response.json() == []


def test_get_companies_returns_list(client: TestClient, session: Session):
    """
    GET /companies returns all companies as a JSON array.
    Inserts two companies and confirms both appear with correct field values.
    Validates API-01: list endpoint returns all rows.
    """
    company_a = _make_company("Airbnb", description="Home sharing platform")
    company_b = _make_company("Stripe", description="Online payment processing")
    session.add(company_a)
    session.add(company_b)
    session.commit()

    response = client.get("/companies/")
    assert response.status_code == 200

    data = response.json()
    assert isinstance(data, list)
    assert len(data) == 2

    names = {c["company_name"] for c in data}
    assert names == {"Airbnb", "Stripe"}

    # Verify descriptions survived round-trip
    descriptions = {c["company_name"]: c["description"] for c in data}
    assert descriptions["Airbnb"] == "Home sharing platform"
    assert descriptions["Stripe"] == "Online payment processing"


# ---------------------------------------------------------------------------
# API-02: GET /companies/{id}
# ---------------------------------------------------------------------------


def test_get_company_by_id(client: TestClient, session: Session):
    """
    GET /companies/{id} returns the correct company record.
    Validates API-02: single-record retrieval by primary key.
    """
    company = _make_company("Dropbox", description="Cloud file storage")
    session.add(company)
    session.commit()
    session.refresh(company)

    assert company.id is not None, "Company must have an id after commit+refresh"

    response = client.get(f"/companies/{company.id}")
    assert response.status_code == 200

    data = response.json()
    assert data["id"] == company.id
    assert data["company_name"] == "Dropbox"
    assert data["description"] == "Cloud file storage"


def test_get_company_not_found(client: TestClient):
    """
    GET /companies/9999 returns HTTP 404 with the standard detail message.
    Validates API-02: 404 behaviour for non-existent primary key.
    """
    response = client.get("/companies/9999")
    assert response.status_code == 404
    assert response.json() == {"detail": "Company not found"}


def test_get_company_invalid_id(client: TestClient):
    """
    GET /companies/abc returns HTTP 422 (Unprocessable Entity).
    FastAPI validates path parameters declared as `int` automatically —
    no custom validation code is required in the route.
    Validates API-02 edge case: non-integer path param handling.
    """
    response = client.get("/companies/abc")
    assert response.status_code == 422


# ---------------------------------------------------------------------------
# API-01 + API-04: AI fields present in response (even when null)
# ---------------------------------------------------------------------------


def test_get_companies_includes_ai_fields(client: TestClient, session: Session):
    """
    GET /companies response objects include all four AI-generated fields.
    Fields may be null (not yet processed by Phase 3 analyser) but must
    be present as keys in the JSON — they must not be omitted.

    Validates:
      API-01: all AI-generated fields appear in the list response
      API-04: Pydantic v2 / SQLModel serialises Optional fields as null,
              not by omitting the key
    """
    company = _make_company(
        "Notion",
        description="All-in-one workspace",
        # AI fields intentionally left as None (default) — mirrors real DB state
    )
    session.add(company)
    session.commit()

    response = client.get("/companies/")
    assert response.status_code == 200

    data = response.json()
    assert len(data) == 1

    record = data[0]
    ai_fields = ["industry", "business_model", "summary", "use_case"]
    for field in ai_fields:
        assert field in record, f"AI field '{field}' missing from response — must be present as null, not omitted"

    # With no Phase 3 processing, all AI fields should be null
    for field in ai_fields:
        assert record[field] is None, (
            f"Expected AI field '{field}' to be null for unprocessed company, got {record[field]!r}"
        )
```
